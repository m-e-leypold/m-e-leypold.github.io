COMPILER       = gcc
EXTRA_CXXFLAGS = -std=c++0x

# clang on my debian 6.x is old enough not to support -MD. FTM I therefore can't use clang
# there. A more modern compiler would have to use -std=c++11.

include $(shell mklib cxx-generic.mklib)

PANDOC-DOCUMENTS = $(patsubst %.md, %, $(wildcard *.md))
PANDOC-RENDERED  = $(PANDOC-DOCUMENTS:%=%.pdf)

pandoc-all: $(PANDOC-RENDERED)

%.pdf: %.md
	markdown2pdf $<

clean:: 
	rm -f *.pdf

/*
 * Demonstrates automatic conversions for rvalue references: The only forbidden
 * conversion is that of T&& to a non const T&.
 *
 */

#include <vector>
#include <iostream>

using namespace std;

class Multiples : public vector<int> { 
public:
  Multiples(int count, int step){
    for ( int i = step; i <= step*count; i = i + step ){
      push_back(i);
    }
  }
  Multiples(){};
};

ostream& 
operator << (ostream& strm, const Multiples& m) {

  strm << "Multiples @ " << &m << " counting " << m.size() << ":" <<endl;

  for (unsigned int i=0; i<m.size(); i++ ) { strm << " " << m[i]; } 

  strm << "." << endl;
  return strm;
};

Multiples
multiples_of_7(int count){
  Multiples m(count,7);
  return m;
};

void as_rvalue_ref( Multiples&& m ){
  cout << __FUNCTION__ << ": received as Multiples&&:" << endl;
  cout << m;
}

void as_const_rvalue_ref( const Multiples&& m ){
  cout << __FUNCTION__ << ": received as const Multiples&&:" << endl;
  cout << m;
}

void as_const_ref( const Multiples& m ){
  cout << __FUNCTION__ << ": received as const Multiples&:" << endl;
  cout << m;
}

void as_ref( Multiples& m ){
  cout << __FUNCTION__ << ": received as Multiples&:" << endl;
  cout << m;
}

int main(){

  as_const_rvalue_ref( multiples_of_7(4) );
  as_rvalue_ref( multiples_of_7(4) );
  as_const_ref( multiples_of_7(4) );
  // as_ref( multiples_of_7(4) );      // this won't fly

  Multiples m(3,3);
  as_const_rvalue_ref( m );
  as_rvalue_ref( m );
  as_const_ref( m );
  as_ref( m );

  return 0;
}

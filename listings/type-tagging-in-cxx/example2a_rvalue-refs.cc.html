// A short demonstration, what C++ &&-References are good for. This needs to be extended
// for the article.

#include <iostream>
#include <vector>

using namespace std;

vector<string> make_vector(){
  vector<string> v;
  v.push_back("uno");
  v.push_back("due");
  v.push_back("tre");
  return v;
}


void dump_vector( vector<string>& v ){
  for ( vector<string>::iterator i =  v.begin(); i!=v.end(); ++i){
    cout << (*i) << endl;
  }
}

void print_vector( vector<string>& v ){  // (1)
  cout << "vector<string>& =>" << endl;
  dump_vector(v);
}

void print_vector( vector<string>&& v ){ // (2)
  cout << "vector<string>&& =>" << endl;
  dump_vector(v);
}


int main(){
  
  int i = 0;
   
  cout << "starting now ..." << endl;

  vector<string> v1 = make_vector();    // This calls (1)
  print_vector(v1);
  print_vector(make_vector());          // This calls (2) :-), would typically move instead of copy

  cout << ++i;
  
  cout << "OK." << endl;
  return 0;
}

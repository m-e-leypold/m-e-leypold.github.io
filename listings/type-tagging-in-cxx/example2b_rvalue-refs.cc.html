/*
  Demonstrates type tagging.
*/

#include <vector>
#include <iostream>

using namespace std;

class Multiples : public vector<int> { 
public:
  Multiples(int count, int step){
    for ( int i = step; i <= step*count; i = i + step ){
      push_back(i);
    }
  }
  Multiples(){};
};

ostream& 
operator << (ostream& strm, Multiples& m) {

  strm << "Multiples @ " << &m << " counting " << m.size() << ":" <<endl;

  for (unsigned int i=0; i<m.size(); i++ ) { strm << " " << m[i]; } 

  strm << "." << endl;
  return strm;
};

Multiples
multiples_of_7(int count){
  Multiples m(count,7);
  return m;
};

class SomeThing {
public:
  SomeThing(Multiples&& dm) {
    cout << "Something: Received disposable argument - can swap/move." << endl;
    swap(this->m, dm);
  }
  SomeThing(const Multiples& m) {
    cout << "Something: Received NON-disposable argument - need to copy" << endl;
    this->m =  m;
  }
private:
  Multiples m;
};

class SomeThingElse {
public:
  SomeThingElse(const Multiples& m){
    cout << "SomethingElse: Treating as NON-disposable argument - need to copy" << endl;
  }
  /* ... */
};

int main(){
  Multiples m(3,7);
  cout << m;

  SomeThing st2( m );
  cout << m;
  SomeThing st1( move(m) );
  cout << m;
  SomeThing st3( multiples_of_7(4) );

  cout << "   ---" << endl;

  Multiples m2(5,5);
  cout << m2;

  SomeThingElse se1( m2 );                    // here it can convert to a Multiples&
  SomeThingElse se2( move(m2) );              // can't convert from Multiples&& to Multiples& 
                                              // but to (const Multiples&)

  SomeThingElse se3( multiples_of_7(4) );     // here too

  cout << m2;

  return 0;
}

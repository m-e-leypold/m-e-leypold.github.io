/*
  Demonstrates type tagging.
*/

#include <vector>
#include <iostream>

using namespace std;

class Multiples : public vector<int> { 
public:
  Multiples(int count, int step){
    for ( int i = step; i <= step*count; i = i + step ){
      push_back(i);
    }
  }
  Multiples(){};
};

ostream& 
operator << (ostream& strm, Multiples& m) {

  strm << "Multiples @ " << &m << " counting " << m.size() << ":" <<endl;

  for (unsigned int i=0; i<m.size(); i++ ) { strm << " " << m[i]; } 

  strm << "." << endl;
  return strm;
};

template<typename T>
class DisposableRef { 
public:
  inline DisposableRef(T& inner) : inner(inner){ };
  inline operator T&() { return inner ; }
  inline T& ref()      { return inner ; }
private:
  T& inner;
};

template<typename T>
inline DisposableRef<T> disposable(T& inner){ 
  return DisposableRef<T>(inner);
}

class SomeThing {
public:
  SomeThing(DisposableRef<Multiples> dm) {
    cout << "Something: Received disposable argument - can swap/move." << endl;
    swap(this->m, dm.ref());
  }
  SomeThing(const Multiples& m) {
    cout << "Something: Received NON-disposable argument - need to copy" << endl;
    this->m =  m;
  }
private:
  Multiples m;
};

class SomeThingElse {
public:
  SomeThingElse(const Multiples& m){
    cout << "SomethingElse: Treating as NON-disposable argument - need to copy" << endl;
  }
  /* ... */
};


int main(){
  Multiples m(3,7);
  cout << m;

  SomeThing st2( m );
  cout << m;

  SomeThing st1( disposable(m) );
  cout << m;

  cout << "   ---" << endl;

  Multiples m2(5,5);
  cout << m2;

  SomeThingElse se2( m2 );
  SomeThingElse se1( disposable(m2) );

  cout << m2;

  return 0;
}

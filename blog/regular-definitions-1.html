from greenland.regexdefinitions import *

rx,ex    = regular_expressions()

ex.DIGS2 = "[0-9]{2}"
ex.DIGS4 = "[0-9]{4}"

rx.YEAR  = "DIGS4"
rx.MONTH = "DIGS2"
rx.DAY   = "DIGS2"
rx.DATE  = "YEAR-MONTH-DAY"

date = rx.DATE.match("2014-09-13")

print( date.fields )
print( "Year : ", date.YEAR )
print( "Month: ", date.MONTH )
print( "Day  : ", date.DAY )
